View code example 
$.fn.coolChangeCols({

    nameCookie : 'demo2-cookie',             // name of cookie required
    activeDefault : 'view2',                 // one value of elements with data-ccc-view-name attribute
    classActiveButton : 'active',            // class of view active into control element
    cookieDays : 1,                          // time of view persist in days
    itemToChangeSelector :'.itemDemoChange', // selector of all items to change view
    effect : true,
    timeEffect : 300,
    
    //events

    afterChangeView : function (view){
        $("#cookieOutput").text(view);
    },

    afterLoadView : function (view){
        $("#cookieOutput").text(view);
    }
});
